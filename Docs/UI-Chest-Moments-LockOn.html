1. Patches — CombatManager + CombatMode

O teu CombatManager já guarda abilityCooldownEndTime mas não o expõe. A stamina BotW precisa disto.

Assets/Scripts/Combat/CombatManager.cs — acrescenta
public float CooldownRemaining =>
    Mathf.Max(0f, abilityCooldownEndTime - Time.time);

public float CooldownDuration
{
    get
    {
        if (TryGetCurrentWeaponAbility(out WeaponAbility w) && w.cooldown > 0f)
            return w.cooldown;
        return 0f;
    }
}

/// <summary>1 = pronto a atacar; 0 = acabou de atacar.</summary>
public float CooldownNormalized
{
    get
    {
        float dur = CooldownDuration;
        if (dur <= 0f) return 1f;
        return 1f - Mathf.Clamp01(CooldownRemaining / dur);
    }
}

public bool IsOnCooldown => Time.time < abilityCooldownEndTime;
Assets/Scripts/UI/CombatMode.cs
using UnityEngine;

namespace AF
{
    /// <summary>
    /// Fonte única: estamos em combate? As world bars de HP usam isto.
    /// </summary>
    public class CombatMode : MonoBehaviour
    {
        [SerializeField] CharacterManager characterManager;
        [SerializeField] float exitDelay = 2f;

        float leaveAt = -1f;
        bool inCombat;

        public bool IsInCombat => inCombat;

        void Update()
        {
            LockOn lockOn = characterManager != null ? characterManager.lockOn : null;
            bool threat = lockOn != null
                && (lockOn.isLockedOn || lockOn.chasingEnemies.Count > 0);

            if (threat)
            {
                inCombat = true;
                leaveAt = -1f;
            }
            else if (inCombat)
            {
                if (leaveAt < 0f) leaveAt = Time.time + exitDelay;
                if (Time.time >= leaveAt) inCombat = false;
            }
        }
    }
}

2. Chest

Assets/Scripts/Interactables/Chest.cs
using UnityEngine;
using UnityEngine.Events;

namespace AF
{
    public class Chest : MonoBehaviour
    {
        [Header("Loot")]
        [SerializeField] Item[] contents;
        [SerializeField] bool openOnce = true;

        [Header("Presentation")]
        [SerializeField] Animator animator;          // opcional — trigger "Open"
        [SerializeField] string openTrigger = "Open";
        [SerializeField] GameObject closedVisual;
        [SerializeField] GameObject openVisual;

        [Header("Events")]
        public UnityEvent onOpened;

        bool opened;

        public bool IsOpened => opened;

        /// <summary>Chamado por um trigger / botão de interact / Moment.</summary>
        public void Open(InventoryManager inventory)
        {
            if (opened && openOnce) return;
            if (inventory == null) return;

            opened = true;

            if (contents != null)
            {
                for (int i = 0; i < contents.Length; i++)
                {
                    if (contents[i] != null)
                        inventory.AddItem(contents[i]);
                }
            }

            if (animator != null && !string.IsNullOrEmpty(openTrigger))
                animator.SetTrigger(openTrigger);

            if (closedVisual != null) closedVisual.SetActive(false);
            if (openVisual != null) openVisual.SetActive(true);

            onOpened?.Invoke();
        }
    }
}
Assets/Scripts/Interactables/ChestInteractTrigger.cs
using UnityEngine;

namespace AF
{
    /// <summary>
    /// Zona de interact. O jogador entra → mostra prompt.
    /// Botão UI / tecla E chama TryOpen().
    /// </summary>
    public class ChestInteractTrigger : MonoBehaviour
    {
        [SerializeField] Chest chest;
        [SerializeField] string prompt = "Abrir baú";

        InventoryManager inventoryInRange;
        public bool CanOpen => inventoryInRange != null && chest != null && !chest.IsOpened;
        public string Prompt => prompt;

        void OnTriggerEnter(Collider other)
        {
            if (other.TryGetComponent(out InventoryManager inv))
                inventoryInRange = inv;
            // se InventoryManager está no CharacterManager:
            else if (other.TryGetComponent(out CharacterManager cm))
                inventoryInRange = cm.GetComponent<InventoryManager>();
        }

        void OnTriggerExit(Collider other)
        {
            if (inventoryInRange != null &&
                (other.GetComponent<InventoryManager>() == inventoryInRange
                 || other.GetComponent<CharacterManager>()?.GetComponent<InventoryManager>() == inventoryInRange))
            {
                inventoryInRange = null;
            }
        }

        public void TryOpen()
        {
            if (!CanOpen) return;
            chest.Open(inventoryInRange);
        }

        void Update()
        {
            // fallback teclado no editor
            if (CanOpen && Input.GetKeyDown(KeyCode.E))
                TryOpen();
        }
    }
}

3. World UI — HP (só em combate)

Setup na cena

Child do personagem: Canvas (World Space) + Image fill de HP. Escala tipica 0.01. O script billboard + liga a Health + CombatMode.

Assets/Scripts/UI/WorldHealthBar.cs
using UnityEngine;
using UnityEngine.UI;

namespace AF
{
    public class WorldHealthBar : MonoBehaviour
    {
        [SerializeField] Health health;
        [SerializeField] CombatMode combatMode;   // no jogador; em inimigos podes deixar null e estar sempre on
        [SerializeField] Image fill;
        [SerializeField] CanvasGroup canvasGroup;
        [SerializeField] bool onlyInCombat = true;
        [SerializeField] bool billboard = true;

        Camera cam;

        void OnEnable()
        {
            if (health != null) health.OnDamaged += HandleDamaged;
        }

        void OnDisable()
        {
            if (health != null) health.OnDamaged -= HandleDamaged;
        }

        void Start()
        {
            cam = Camera.main;
            Refresh(immediate: true);
        }

        void LateUpdate()
        {
            bool show = !onlyInCombat
                || combatMode == null
                || combatMode.IsInCombat;

            if (health != null && health.IsDead) show = false;

            if (canvasGroup != null)
                canvasGroup.alpha = show ? 1f : 0f;

            if (billboard && cam != null && show)
                transform.rotation = Quaternion.LookRotation(transform.position - cam.transform.position);
        }

        void HandleDamaged(DamageResult _) => Refresh(false);

        void Refresh(bool immediate)
        {
            if (health == null || fill == null) return;
            float t = health.Max > 0 ? (float)health.Current / health.Max : 0f;
            fill.fillAmount = t;
        }
    }
}
Inimigos

Reutiliza o mesmo script no inimigo com onlyInCombat = false (ou um CombatMode local que fica true enquanto StateCombat). O billboard mantém a barra legível com a terceira pessoa.

using UnityEngine;
using UnityEngine.UIElements;

namespace AF
{
    /// <summary>
    /// World-space UI Toolkit. Coloca UIDocument num child do player
    /// (acima da cabeça / perto da arma). Panel Settings → Render Mode = World Space
    /// (Unity 6 / UI Toolkit world). Em versões sem world UITK, usa um
    /// World Space Canvas + raw Image circular — a lógica CooldownNormalized é igual.
    /// </summary>
    public class WeaponStaminaBar : MonoBehaviour
    {
        [SerializeField] UIDocument document;
        [SerializeField] CombatManager combatManager;
        [SerializeField] CombatMode combatMode;
        [SerializeField] bool hideOutOfCombat = true;
        [SerializeField] bool billboard = true;

        VisualElement root;
        VisualElement fill;
        Label ready;
        Camera cam;

        void OnEnable()
        {
            if (document == null) document = GetComponent<UIDocument>();
            var ve = document.rootVisualElement;
            root = ve.Q("root");
            fill = ve.Q("ring-fill");
            ready = ve.Q<Label>("ready");
            cam = Camera.main;
        }

        void LateUpdate()
        {
            if (combatManager == null || root == null) return;

            bool show = !hideOutOfCombat
                || combatMode == null
                || combatMode.IsInCombat;

            root.style.display = show ? DisplayStyle.Flex : DisplayStyle.None;
            if (!show) return;

            float n = combatManager.CooldownNormalized; // 0..1
            // Feedback visual simples: opacidade + escala do anel
            fill.style.opacity = n;
            float s = Mathf.Lerp(0.75f, 1f, n);
            fill.style.scale = new Scale(new Vector3(s, s, 1f));

            bool readyNow = n >= 0.999f;
            if (ready != null)
                ready.text = readyNow ? "" : "";

            // Cor: amarelo a encher → branco quando pronto (estilo BotW)
            Color c = readyNow
                ? new Color(1f, 1f, 1f, 0.9f)
                : new Color(1f, 0.86f, 0.35f, 0.95f);
            fill.style.borderTopColor = c;
            fill.style.borderRightColor = c;
            fill.style.borderBottomColor = c;
            fill.style.borderLeftColor = c;

            if (billboard && cam != null)
                transform.rotation = Quaternion.LookRotation(
                    transform.position - cam.transform.position);
        }
    }
}
Progresso circular de verdade

UI Toolkit não tem RadialFill nativo. Opções: (1) Image uGUI com fill Method = Radial 360 no World Canvas — o mais fácil e o mais “BotW”; (2) shader / custom painter UITK. Se quiseres o look exacto, usa uGUI Radial para a stamina e UITK só para o menu/diálogo. Abaixo o fallback Radial em 10 linhas.

using System.Collections.Generic;
using UnityEngine;
using UnityEngine.UIElements;

namespace AF
{
    public class GameMenuController : MonoBehaviour
    {
        [SerializeField] UIDocument document;
        [SerializeField] InventoryManager inventory;
        [SerializeField] EquipmentManager equipment;
        [SerializeField] KeyCode toggleKey = KeyCode.Escape;

        static readonly string[] Tabs = { "Itens", "Equipamento", "Skills", "Opções" };

        VisualElement backdrop;
        ListView tabs;
        ListView list;
        Label contentTitle;
        Label detail;
        bool open;
        int tabIndex;

        void OnEnable()
        {
            var root = document.rootVisualElement;
            backdrop = root.Q("backdrop");
            tabs = root.Q<ListView>("tabs");
            list = root.Q<ListView>("list");
            contentTitle = root.Q<Label>("content-title");
            detail = root.Q<Label>("detail");

            tabs.itemsSource = new List<string>(Tabs);
            tabs.makeItem = () => new Label();
            tabs.bindItem = (e, i) => ((Label)e).text = Tabs[i];
            tabs.selectionChanged += _ =>
            {
                if (tabs.selectedIndex >= 0) SelectTab(tabs.selectedIndex);
            };

            list.makeItem = () => new Label();
            list.bindItem = (e, i) =>
            {
                var src = list.itemsSource as List<string>;
                ((Label)e).text = src != null && i < src.Count ? src[i] : "";
            };
            list.selectionChanged += _ => ShowDetail();

            SetOpen(false);
        }

        void Update()
        {
            if (Input.GetKeyDown(toggleKey))
                SetOpen(!open);
        }

        public void SetOpen(bool value)
        {
            open = value;
            if (backdrop != null)
                backdrop.style.display = open ? DisplayStyle.Flex : DisplayStyle.None;

            Time.timeScale = open ? 0f : 1f;

            if (open)
            {
                tabs.SetSelection(0);
                SelectTab(0);
            }
        }

        void SelectTab(int index)
        {
            tabIndex = index;
            contentTitle.text = Tabs[index];
            detail.text = "";

            var rows = new List<string>();

            switch (index)
            {
                case 0: // Itens
                    rows.Add("(vazio por agora — liga consumíveis depois)");
                    break;
                case 1: // Equipamento
                    if (inventory != null)
                    {
                        for (int i = 0; i < inventory.ownedWeapons.Count; i++)
                        {
                            var w = inventory.ownedWeapons[i];
                            bool eq = equipment != null
                                && equipment.weaponInstance != null
                                && equipment.weaponInstance == w;
                            rows.Add((eq ? "E " : "  ") + w.itemData.displayName);
                        }
                    }
                    if (rows.Count == 0) rows.Add("(sem armas)");
                    break;
                case 2: // Skills
                    rows.Add("Ataque automático");
                    rows.Add("Especial");
                    break;
                case 3: // Opções
                    rows.Add("Áudio");
                    rows.Add("Controlos");
                    rows.Add("Voltar ao hub");
                    break;
            }

            list.itemsSource = rows;
            list.Rebuild();
            list.ClearSelection();
        }

        void ShowDetail()
        {
            if (list.selectedIndex < 0) return;
            var src = list.itemsSource as List<string>;
            if (src == null) return;

            string name = src[list.selectedIndex];
            detail.text = name;

            // Equipar arma se tab equipamento
            if (tabIndex == 1 && inventory != null && equipment != null)
            {
                int weaponIndex = list.selectedIndex;
                if (weaponIndex >= 0 && weaponIndex < inventory.ownedWeapons.Count)
                {
                    Weapon weapon = inventory.ownedWeapons[weaponIndex].itemData;
                    equipment.EquipWeapon(weapon);
                    SelectTab(1); // refresh "E " marker
                }
            }
        }
    }
}

using System;
using System.Collections;
using System.Collections.Generic;
using UnityEngine;
using UnityEngine.UIElements;

namespace AF
{
    [Serializable]
    public class DialogueChoice
    {
        public string text;
        public UnityEngine.Events.UnityEvent onChosen;
    }

    public class DialogueUI : MonoBehaviour
    {
        [SerializeField] UIDocument document;
        [SerializeField] float charsPerSecond = 40f;

        VisualElement root;
        Label speaker;
        Label body;
        VisualElement choicesRoot;
        Label continueHint;

        bool skipping;
        bool waitingAdvance;
        int chosenIndex = -1;

        public static DialogueUI Instance { get; private set; }

        void Awake()
        {
            Instance = this;
            var ve = document.rootVisualElement;
            root = ve.Q("root");
            speaker = ve.Q<Label>("speaker");
            body = ve.Q<Label>("body");
            choicesRoot = ve.Q("choices");
            continueHint = ve.Q<Label>("continue");
            Hide();
        }

        void Update()
        {
            // tap / click / E / Space = advance / skip typewriter
            bool pressed = Input.GetMouseButtonDown(0)
                || Input.GetKeyDown(KeyCode.E)
                || Input.GetKeyDown(KeyCode.Space)
                || (Input.touchCount > 0 && Input.GetTouch(0).phase == TouchPhase.Began);

            if (!pressed) return;
            if (root.style.display == DisplayStyle.None) return;

            if (!waitingAdvance)
                skipping = true;
            else
                waitingAdvance = false;
        }

        public IEnumerator PlayLine(string speakerName, string message, IList<DialogueChoice> choices = null)
        {
            root.style.display = DisplayStyle.Flex;
            speaker.text = speakerName ?? "";
            choicesRoot.Clear();
            continueHint.style.display = DisplayStyle.None;
            skipping = false;
            waitingAdvance = false;
            chosenIndex = -1;

            // typewriter
            body.text = "";
            float delay = 1f / Mathf.Max(1f, charsPerSecond);
            for (int i = 0; i < message.Length; i++)
            {
                if (skipping)
                {
                    body.text = message;
                    break;
                }
                body.text += message[i];
                yield return new WaitForSecondsRealtime(delay);
            }

            if (choices != null && choices.Count > 0)
            {
                continueHint.style.display = DisplayStyle.None;
                for (int i = 0; i < choices.Count; i++)
                {
                    int captured = i;
                    var btn = new Button(() => chosenIndex = captured) { text = choices[i].text };
                    btn.AddToClassList("dlg-choice");
                    choicesRoot.Add(btn);
                }

                yield return new WaitUntil(() => chosenIndex >= 0);
                choices[chosenIndex].onChosen?.Invoke();
            }
            else
            {
                continueHint.style.display = DisplayStyle.Flex;
                waitingAdvance = true;
                // wait one frame so o mesmo tap que skipou o typewriter não avança
                yield return null;
                yield return new WaitUntil(() => !waitingAdvance);
            }

            Hide();
        }

        void Hide()
        {
            root.style.display = DisplayStyle.None;
            choicesRoot.Clear();

using System.Collections;
using System.Collections.Generic;
using UnityEngine;
using UnityEngine.Events;

namespace AF
{
    /// <summary>
    /// Inspirado no Moment do Cacildes Adventure 2:
    /// filhos com EventBase, executados em ordem de sibling.
    /// </summary>
    public class Moment : MonoBehaviour
    {
        [TextArea] public string comment = "Filhos = EventBase. Chama Trigger().";

        public UnityEvent onMomentStart;
        public UnityEvent onMomentEnd;

        readonly List<EventBase> events = new();
        Coroutine running;
        public bool IsRunning { get; private set; }

        void Collect()
        {
            events.Clear();
            foreach (Transform child in transform)
            {
                foreach (var ev in child.GetComponents<EventBase>())
                    events.Add(ev);
            }
        }

        public void Trigger()
        {
            if (IsRunning) return;
            Collect();
            onMomentStart?.Invoke();
            MomentRunner.NotifyStart(this);
            running = StartCoroutine(Run());
        }

        IEnumerator Run()
        {
            IsRunning = true;
            for (int i = 0; i < events.Count; i++)
            {
                if (events[i] != null)
                    yield return events[i].Dispatch();
            }
            Finish();
        }

        public void Cancel()
        {
            if (running != null) StopCoroutine(running);
            Finish();
        }

        void Finish()
        {
            running = null;
            IsRunning = false;
            MomentRunner.NotifyEnd(this);
            onMomentEnd?.Invoke();
        }

        void OnDisable()
        {
            if (IsRunning) Finish();
        }
    }

    /// <summary>Flag global leve — sem Event Bus.</summary>
    public static class MomentRunner
    {
        public static bool HasMomentOngoing { get; private set; }
        static Moment current;

        public static void NotifyStart(Moment m)
        {
            current = m;
            HasMomentOngoing = true;
        }

        public static void NotifyEnd(Moment m)
        {
            if (current == m)
            {
                current = null;
                HasMomentOngoing = false;
            }
        }
    }
}

8. Lock-On mobile — tap no inimigo

O que muda

Hoje o LockOn escolhe automaticamente o inimigo mais próximo da lista chasingEnemies. Passa a: tap no ecrã → raycast → se for inimigo, lock. A lista de chase continua a servir para limpar mortos/longe e para o CombatMode, mas não escolhe o alvo sozinha.

Assets/Scripts/Camera/LockOn.cs — versão nova
using System.Collections.Generic;
using UnityEngine;

namespace AF
{
    public class LockOn : MonoBehaviour
    {
        [Header("Lock On")]
        public Transform lockOnTarget;
        public bool isLockedOn;
        public float lockOnRotationSpeed = 15f;
        public float lockOnMaxDistance = 12f;

        [Header("Threat Tracking")]
        public List<CharacterManager> chasingEnemies = new();

        [SerializeField] CharacterManager characterManager;

        public void RegisterChasingEnemy(CharacterManager enemy)
        {
            if (enemy != null && !chasingEnemies.Contains(enemy))
                chasingEnemies.Add(enemy);
        }

        public void UnregisterChasingEnemy(CharacterManager enemy)
        {
            chasingEnemies.Remove(enemy);
            if (lockOnTarget != null && enemy != null
                && lockOnTarget == enemy.transform)
                ClearLockOn();
        }

        void Update()
        {
            if (!characterManager.IsPlayer()) return;
            PruneThreats();

            // Se o alvo actual morreu / saiu de alcance → clear
            if (isLockedOn && !IsValidTarget(lockOnTarget))
                ClearLockOn();
        }

        void PruneThreats()
        {
            float maxSqr = lockOnMaxDistance * lockOnMaxDistance;
            for (int i = chasingEnemies.Count - 1; i >= 0; i--)
            {
                var e = chasingEnemies[i];
                if (e == null || e.health.IsDead)
                {
                    chasingEnemies.RemoveAt(i);
                    continue;
                }

                float sqr = (e.transform.position - transform.position).sqrMagnitude;
                if (sqr > maxSqr) chasingEnemies.RemoveAt(i);
            }
        }

        public bool TryLockOn(CharacterManager enemy)
        {
            if (enemy == null || enemy.health.IsDead) return false;

            float sqr = (enemy.transform.position - transform.position).sqrMagnitude;
            if (sqr > lockOnMaxDistance * lockOnMaxDistance) return false;

            RegisterChasingEnemy(enemy);
            SetLockOn(enemy.transform);
            return true;
        }

        public void SetLockOn(Transform target)
        {
            lockOnTarget = target;
            isLockedOn = target != null;
        }

        public void ClearLockOn()
        {
            lockOnTarget = null;
            isLockedOn = false;
        }

        bool IsValidTarget(Transform t)
        {
            if (t == null) return false;
            var cm = t.GetComponent<CharacterManager>();
            if (cm == null || cm.health.IsDead) return false;
            float sqr = (t.position - transform.position).sqrMagnitude;
            return sqr <= lockOnMaxDistance * lockOnMaxDistance;
        }
    }
}
Assets/Scripts/Camera/TapLockOnInput.cs
using UnityEngine;
using UnityEngine.EventSystems;

namespace AF
{
    /// <summary>
    /// Mobile: tap no inimigo = lock. Tap no chão / nada = clear.
    /// Ignora taps sobre UI (botões, joystick).
    /// </summary>
    public class TapLockOnInput : MonoBehaviour
    {
        [SerializeField] LockOn lockOn;
        [SerializeField] Camera cam;
        [SerializeField] LayerMask enemyLayer;     // layer dos inimigos / DamageReceiver
        [SerializeField] LayerMask groundMask;     // opcional
        [SerializeField] float maxRayDistance = 40f;
        [SerializeField] float tapMoveThreshold = 20f; // px — distingue tap de drag do look/joystick

        Vector2 pressPos;
        int fingerId = -1;

        void Start()
        {
            if (cam == null) cam = Camera.main;
        }

        void Update()
        {
            if (lockOn == null || MomentRunner.HasMomentOngoing) return;
            if (Time.timeScale <= 0f) return; // menu aberto

#if UNITY_EDITOR
            HandleMouse();
#endif
            HandleTouches();
        }

        void HandleTouches()
        {
            for (int i = 0; i < Input.touchCount; i++)
            {
                Touch t = Input.GetTouch(i);

                if (t.phase == TouchPhase.Began)
                {
                    if (IsOverUI(t.fingerId)) continue;
                    fingerId = t.fingerId;
                    pressPos = t.position;
                }
                else if (t.fingerId == fingerId
                         && (t.phase == TouchPhase.Ended || t.phase == TouchPhase.Canceled))
                {
                    float moved = (t.position - pressPos).magnitude;
                    if (moved <= tapMoveThreshold)
                        ProcessTap(t.position);
                    fingerId = -1;
                }
            }
        }

#if UNITY_EDITOR
        void HandleMouse()
        {
            if (Input.touchCount > 0) return;

            if (Input.GetMouseButtonDown(0))
            {
                if (IsOverUI(-1)) return;
                pressPos = Input.mousePosition;
            }

            if (Input.GetMouseButtonUp(0))
            {
                float moved = ((Vector2)Input.mousePosition - pressPos).magnitude;
                if (moved <= tapMoveThreshold)
                    ProcessTap(Input.mousePosition);
            }
        }
#endif

        void ProcessTap(Vector2 screenPos)
        {
            Ray ray = cam.ScreenPointToRay(screenPos);

            if (Physics.Raycast(ray, out RaycastHit hit, maxRayDistance, enemyLayer,
                    QueryTriggerInteraction.Ignore))
            {
                CharacterManager enemy = hit.collider.GetComponentInParent<CharacterManager>();
                if (enemy != null && !enemy.IsPlayer())
                {
                    // Tap no mesmo alvo = toggle off
                    if (lockOn.isLockedOn && lockOn.lockOnTarget == enemy.transform)
                        lockOn.ClearLockOn();
                    else
                        lockOn.TryLockOn(enemy);
                    return;
                }
            }

            // Tap no vazio → clear
            lockOn.ClearLockOn();
        }

        static bool IsOverUI(int id)
        {
            if (EventSystem.current == null) return false;
            return id >= 0
                ? EventSystem.current.IsPointerOverGameObject(id)
                : EventSystem.current.IsPointerOverGameObject();
        }
    }
}
Layer tip

Mete os inimigos numa layer Enemy e selecciona-a em enemyLayer. O raycast não precisa de bater no mesh exacto — um capsule collider no root chega. O joystick / botões UI são ignorados via EventSystem.IsPointerOverGameObject.
