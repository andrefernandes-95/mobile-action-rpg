0. Mapa + decisões

Modelo de jogo
  • Soulslike — morte → largar todo o ouro como bloodstain (único; o novo apaga o anterior) → respawn no último bonfire (cura + repor poções). Tocar o bloodstain recupera o ouro.
  • Armas consumíveis — nunca permanentes entre runs. Durabilidade gasta por golpe acertado; a 0 a arma parte e sai do inventário.
  • Ferreiro — reparar e estender uso. Reparar é sempre mais barato que recomprar a mesma arma.
  • Loja — o ferreiro/comerciante vende armas (e poções). Ouro ganho ao matar inimigos.
  • Menu — gear icon no ecrã. Tabs: Inventário · Loja · Ferreiro · Opções. Look FF simplificado (UI Toolkit).
  • Diálogos — estilo Final Fantasy: portrait + caixa; mensagens importantes com fundo escuro / sem fundo, Top / Center / Bottom. Data-driven (ScriptableObjects).
  • Cutscenes — SequencePlayer com steps assíncronos (substitui Moment+EventBase do doc antigo). Sem Timeline, sem Event Bus global.

1. Economia — Wallet + GoldReward

Conceito

Wallet vive dentro de PlayerProgress (secção 2). Inimigos têm GoldReward que credita ouro em health.OnDied.

Assets/Scripts/Economy/Wallet.cs
using System;
using UnityEngine;

namespace AF
{
    [Serializable]
    public class Wallet
    {
        [SerializeField] int gold;

        public int Gold => gold;

        public event Action<int> OnGoldChanged;

        public void Add(int amount)
        {
            if (amount <= 0) return;
            gold += amount;
            OnGoldChanged?.Invoke(gold);
        }

        public bool TrySpend(int amount)
        {
            if (amount < 0 || gold < amount) return false;
            gold -= amount;
            OnGoldChanged?.Invoke(gold);
            return true;
        }

        /// <summary>Zera e devolve o valor anterior (usado no bloodstain).</summary>
        public int DrainAll()
        {
            int taken = gold;
            gold = 0;
            if (taken > 0) OnGoldChanged?.Invoke(gold);
            return taken;
        }

        public void Set(int amount)
        {
            gold = Mathf.Max(0, amount);
            OnGoldChanged?.Invoke(gold);
        }
    }
}
Assets/Scripts/Economy/GoldReward.cs
using UnityEngine;

namespace AF
{
    /// <summary>
    /// Coloca no inimigo. Ao morrer, credita ouro ao Wallet do jogador.
    /// </summary>
    public class GoldReward : MonoBehaviour
    {
        [SerializeField] Health health;
        [SerializeField] int goldAmount = 25;
        [SerializeField] int goldVariance = 5; // ±

        bool paid;

        void OnEnable()
        {
            if (health != null) health.OnDied += HandleDied;
        }

        void OnDisable()
        {
            if (health != null) health.OnDied -= HandleDied;
        }

        void HandleDied()
        {
            if (paid) return;
            paid = true;

            if (PlayerProgress.Instance == null) return;

            int lo = Mathf.Max(0, goldAmount - goldVariance);
            int hi = goldAmount + goldVariance;
            int amount = Random.Range(lo, hi + 1);
            PlayerProgress.Instance.Wallet.Add(amount);
        }
    }
}

2. Progressão — Bonfire · Bloodstain · DeathReactor

Fluxo soulslike

PlayerProgress (DontDestroyOnLoad) guarda Wallet, último bonfire e referência ao bloodstain actual. DeathReactor deixa de fazer LoadScene — passa a largar bloodstain e respawnar.

Assets/Scripts/Progression/PlayerProgress.cs
using UnityEngine;

namespace AF
{
    public class PlayerProgress : MonoBehaviour
    {
        public static PlayerProgress Instance { get; private set; }

        public Wallet Wallet = new();

        [SerializeField] Bonfire lastBonfire;
        [SerializeField] Bloodstain bloodstainPrefab;

        Bloodstain currentBloodstain;

        public Bonfire LastBonfire => lastBonfire;

        void Awake()
        {
            if (Instance != null && Instance != this)
            {
                Destroy(gameObject);
                return;
            }
            Instance = this;
            DontDestroyOnLoad(gameObject);
        }

        public void SetBonfire(Bonfire bonfire)
        {
            lastBonfire = bonfire;
        }

        public Bloodstain DropBloodstain(Vector3 worldPos)
        {
            int dropped = Wallet.DrainAll();
            if (dropped <= 0) return null;

            if (currentBloodstain != null)
                Destroy(currentBloodstain.gameObject);

            if (bloodstainPrefab == null)
            {
                Debug.LogWarning("Bloodstain prefab em falta.");
                return null;
            }

            currentBloodstain = Instantiate(bloodstainPrefab, worldPos, Quaternion.identity);
            currentBloodstain.Init(dropped);
            return currentBloodstain;
        }

        public void ClearBloodstain(Bloodstain stain)
        {
            if (currentBloodstain == stain)
                currentBloodstain = null;
        }
    }
}
Assets/Scripts/Progression/Bonfire.cs
using UnityEngine;
using UnityEngine.Events;

namespace AF
{
    public class Bonfire : MonoBehaviour
    {
        [SerializeField] Transform respawnPoint;
        [SerializeField] string prompt = "Descansar na fogueira";
        [SerializeField] int potionRestockAmount = 3;

        public UnityEvent onRested;

        public Vector3 RespawnPosition =>
            respawnPoint != null ? respawnPoint.position : transform.position;

        public Quaternion RespawnRotation =>
            respawnPoint != null ? respawnPoint.rotation : transform.rotation;

        /// <summary>Chamado por interact / botão UI.</summary>
        public void Rest(CharacterManager player)
        {
            if (player == null) return;

            PlayerProgress.Instance?.SetBonfire(this);

            // Cura total
            if (player.health != null)
                player.health.SetMax(player.health.Max, refill: true);

            // Repõe poções
            var inv = player.GetComponent<InventoryManager>();
            inv?.RestockPotions(potionRestockAmount);

            onRested?.Invoke();
        }

        // Fallback teclado no editor (zona = trigger no mesmo GO ou child)
        InventoryManager playerInRange;

        void OnTriggerEnter(Collider other)
        {
            if (other.TryGetComponent(out CharacterManager cm) && cm.IsPlayer())
                playerInRange = cm.GetComponent<InventoryManager>();
        }

        void OnTriggerExit(Collider other)
        {
            if (other.TryGetComponent(out CharacterManager cm) && cm.IsPlayer())
                playerInRange = null;
        }

        void Update()
        {
            if (playerInRange != null && Input.GetKeyDown(KeyCode.E))
            {
                var cm = playerInRange.GetComponent<CharacterManager>();
                Rest(cm);
            }
        }
    }
}
Assets/Scripts/Progression/Bloodstain.cs
using UnityEngine;

namespace AF
{
    public class Bloodstain : MonoBehaviour
    {
        [SerializeField] int goldHeld;
        [SerializeField] float pickupRadius = 1.5f;

        public void Init(int gold)
        {
            goldHeld = gold;
        }

        void Update()
        {
            var player = GameContext.Player;
            if (player == null || goldHeld <= 0) return;

            float sqr = (player.transform.position - transform.position).sqrMagnitude;
            if (sqr > pickupRadius * pickupRadius) return;

            PlayerProgress.Instance?.Wallet.Add(goldHeld);
            PlayerProgress.Instance?.ClearBloodstain(this);
            goldHeld = 0;
            Destroy(gameObject);
        }
    }
}

void HandleDeath()
{
    if (characterManager == null) return;

    characterManager.animator.Play(AnimHashes.Death);
    characterManager.Stop();

    if (characterManager.Motor != null)
        characterManager.Motor.SetMotorEnabled(false);

    characterManager.isBusy = true;

    if (characterManager.IsPlayer())
        StartCoroutine(SoulslikeRespawn());
}

IEnumerator SoulslikeRespawn()
{
    yield return new WaitForSeconds(2f);

    // 1) Larga bloodstain com todo o ouro
    PlayerProgress.Instance?.DropBloodstain(characterManager.transform.position);

    // 2) Respawn no último bonfire (ou posição actual se ainda não houve rest)
    Bonfire bf = PlayerProgress.Instance != null
        ? PlayerProgress.Instance.LastBonfire
        : null;

    if (bf != null)
    {
        characterManager.transform.SetPositionAndRotation(
            bf.RespawnPosition, bf.RespawnRotation);
    }

    // 3) Cura + reactivar
    if (characterManager.health != null)
        characterManager.health.SetMax(characterManager.health.Max, refill: true);

    if (characterManager.Motor != null)
        characterManager.Motor.SetMotorEnabled(true);

    characterManager.isBusy = false;
    characterManager.animator.Play(AnimHashes.Idle); // ou o hash de idle que tiveres
}

Assets/Scripts/Items/WeaponInstance.cs — substitui / amplia a classe
namespace AF
{
    public class WeaponInstance : ItemInstance
    {
        public Weapon weaponData;
        public WeaponAbility ability;

        public int durability;
        public int maxDurability; // pode subir via ferreiro (estender)

        public bool IsBroken => durability <= 0;
        public float DurabilityNormalized =>
            maxDurability > 0 ? (float)durability / maxDurability : 0f;

        public WeaponInstance(string id, Weapon weaponData) : base(id, weaponData)
        {
            this.weaponData = weaponData;
            this.ability = new WeaponAbility(
                weaponData.damageType, weaponData.amount, weaponData.engageRadius);

            maxDurability = weaponData.maxDurability;
            durability = maxDurability;
        }

        public void ConsumeDurability(int amount = 1)
        {
            if (amount <= 0) return;
            durability = System.Math.Max(0, durability - amount);
        }

        public void Repair()
        {
            durability = maxDurability;
        }

        public void Extend(int points)
        {
            if (points <= 0) return;
            maxDurability += points;
            durability += points; // também enche o extra
        }

        public int RepairCost =>
            (maxDurability - durability) * weaponData.repairCostPerPoint;

        public int ExtendCost =>
            weaponData.extendAmount * weaponData.extendCostPerPoint;
    }
}
Assets/Scripts/Combat/WeaponDurability.cs
using System;
using UnityEngine;

namespace AF
{
    /// <summary>
    /// No player. Cada onEnemyHit consome 1 durabilidade da arma equipada.
    /// </summary>
    public class WeaponDurability : MonoBehaviour
    {
        [SerializeField] CombatManager combatManager;
        [SerializeField] EquipmentManager equipmentManager;
        [SerializeField] InventoryManager inventoryManager;

        public event Action<WeaponInstance> OnWeaponBroke;

        void OnEnable()
        {
            if (combatManager != null)
                combatManager.onEnemyHit.AddListener(HandleHit);
        }

        void OnDisable()
        {
            if (combatManager != null)
                combatManager.onEnemyHit.RemoveListener(HandleHit);
        }

        void HandleHit(Vector3 _)
        {
            WeaponInstance wi = equipmentManager != null
                ? equipmentManager.weaponInstance
                : null;
            if (wi == null) return;

            wi.ConsumeDurability(1);
            if (!wi.IsBroken) return;

            BreakWeapon(wi);
        }

        void BreakWeapon(WeaponInstance wi)
        {
            equipmentManager.UnequipWeapon();
            inventoryManager.ownedWeapons.Remove(wi);
            OnWeaponBroke?.Invoke(wi);

            // Opcional: auto-equipar a primeira arma restante
            if (inventoryManager.ownedWeapons.Count > 0)
            {
                Weapon next = inventoryManager.ownedWeapons[0].weaponData;
                equipmentManager.EquipWeapon(next);
            }
        }
    }
}

using UnityEngine;

namespace AF
{
    public class ShopService : MonoBehaviour
    {
        [SerializeField] ShopInventory catalog;
        [SerializeField] InventoryManager inventory;

        public ShopInventory Catalog => catalog;

        public bool BuyWeapon(Weapon weapon)
        {
            if (weapon == null || inventory == null) return false;
            if (PlayerProgress.Instance == null) return false;

            int price = weapon.buyPrice;
            if (!PlayerProgress.Instance.Wallet.TrySpend(price)) return false;

            inventory.AddItem(weapon);
            return true;
        }

        public bool BuyConsumable(Consumable item, int count = 1)
        {
            if (item == null || inventory == null || count <= 0) return false;
            if (PlayerProgress.Instance == null) return false;

            int total = item.buyPrice * count;
            if (!PlayerProgress.Instance.Wallet.TrySpend(total)) return false;

            for (int i = 0; i < count; i++)
                inventory.AddItem(item);
            return true;
        }
    }
}

namespace AF
{
    public class BlacksmithService : MonoBehaviour
    {
        public bool Repair(WeaponInstance instance)
        {
            if (instance == null || instance.durability >= instance.maxDurability)
                return false;
            if (PlayerProgress.Instance == null) return false;

            int cost = instance.RepairCost;
            if (cost <= 0) return false;
            if (!PlayerProgress.Instance.Wallet.TrySpend(cost)) return false;

            instance.Repair();
            return true;
        }

        public bool Extend(WeaponInstance instance)
        {
            if (instance == null || instance.weaponData == null) return false;
            if (PlayerProgress.Instance == null) return false;

            int cost = instance.ExtendCost;
            if (!PlayerProgress.Instance.Wallet.TrySpend(cost)) return false;

            instance.Extend(instance.weaponData.extendAmount);
            return true;
        }
    }
}

Assets/Scripts/Items/InventoryManager.cs — amplia
using System;
using System.Collections.Generic;
using System.Linq;
using UnityEngine;

namespace AF
{
    [Serializable]
    public class ConsumableStack
    {
        public Consumable item;
        public int count;
    }

    public class InventoryManager : MonoBehaviour
    {
        public List<WeaponInstance> ownedWeapons = new();
        public List<ConsumableStack> consumables = new();

        [Header("Default Equipment")]
        [SerializeField] Weapon defaultWeapon;
        [SerializeField] Weapon[] randomDefaultWeapons;

        [Header("Default Consumables")]
        [SerializeField] Consumable defaultPotion;
        [SerializeField] int defaultPotionCount = 3;

        [Header("Components")]
        [SerializeField] EquipmentManager equipmentManager;
        [SerializeField] Health health; // player health para UseConsumable

        void Awake()
        {
            if (randomDefaultWeapons.Length > 0)
            {
                Weapon selected = randomDefaultWeapons[
                    UnityEngine.Random.Range(0, randomDefaultWeapons.Length)];
                AddItem(selected);
                selected.Equip(equipmentManager);
            }
            else if (defaultWeapon != null)
            {
                AddItem(defaultWeapon);
                defaultWeapon.Equip(equipmentManager);
            }

            if (defaultPotion != null && defaultPotionCount > 0)
            {
                for (int i = 0; i < defaultPotionCount; i++)
                    AddItem(defaultPotion);
            }
        }

        public void AddItem<T>(T item) where T : Item
        {
            if (item is Weapon weapon)
            {
                string id = Guid.NewGuid().ToString();
                ownedWeapons.Add(new WeaponInstance(id, weapon));
                return;
            }

            if (item is Consumable consumable)
            {
                var stack = consumables.FirstOrDefault(s => s.item == consumable);
                if (stack != null) stack.count++;
                else consumables.Add(new ConsumableStack { item = consumable, count = 1 });
            }
        }

        public bool UseConsumable(Consumable item)
        {
            var stack = consumables.FirstOrDefault(s => s.item == item);
            if (stack == null || stack.count <= 0) return false;
            if (health == null || health.IsDead) return false;

            health.Heal(item.healAmount);
            stack.count--;
            if (stack.count <= 0) consumables.Remove(stack);
            return true;
        }

        public void RestockPotions(int amount)
        {
            // Repõe stacks com restockAtBonfire até amount (ou defaultRestockCount)
            foreach (var stack in consumables)
            {
                if (stack.item != null && stack.item.restockAtBonfire)
                {
                    int target = amount > 0 ? amount : stack.item.defaultRestockCount;
                    stack.count = Mathf.Max(stack.count, target);
                }
            }

            // Se não há nenhuma poção, e tens defaultPotion, cria stack
            if (defaultPotion != null && defaultPotion.restockAtBonfire)
            {
                var stack = consumables.FirstOrDefault(s => s.item == defaultPotion);
                int target = amount > 0 ? amount : defaultPotion.defaultRestockCount;
                if (stack == null)
                    consumables.Add(new ConsumableStack { item = defaultPotion, count = target });
                else
                    stack.count = Mathf.Max(stack.count, target);
            }
        }

        public bool TryGetWeaponInstance(Weapon weapon, out WeaponInstance weaponInstance)
        {
            weaponInstance = ownedWeapons.FirstOrDefault(
                ownedWeapon => ownedWeapon.itemData.Equals(weapon));
            return weaponInstance != null;
        }
    }
}

namespace AF

    public class GameMenuController : MonoBehaviour
    {
        [SerializeField] UIDocument document;
        [SerializeField] InventoryManager inventory;
        [SerializeField] EquipmentManager equipment;
        [SerializeField] ShopService shop;
        [SerializeField] BlacksmithService blacksmith;

        static readonly string[] Tabs = { "Inventário", "Loja", "Ferreiro", "Opções" };

        VisualElement backdrop;
        Button gearBtn;
        Button closeBtn;
        ListView tabs;
        ListView list;
        Label contentTitle;

        VisualElement actions;

        bool open;

        void OnEnable()
        {
            var root = document.rootVisualElement;
            backdrop = root.Q("backdrop");
            gearBtn = root.Q<Button>("gear-btn");
            closeBtn = root.Q<Button>("close-btn");
            tabs = root.Q<ListView>("tabs");
            list = root.Q<ListView>("list");
            contentTitle = root.Q<Label>("content-title");
            detail = root.Q<Label>("detail");
            goldLabel = root.Q<Label>("gold-label");
            goldHud = root.Q<Label>("gold-hud");
            actions = root.Q("actions");

            gearBtn.clicked += () => SetOpen(!open);
            closeBtn.clicked += () => SetOpen(false);

            tabs.itemsSource = new List<string>(Tabs);
            tabs.makeItem = () => new Label();
            tabs.bindItem = (e, i) => ((Label)e).text = Tabs[i];
            tabs.selectionChanged += _ =>
            {
                if (tabs.selectedIndex >= 0) SelectTab(tabs.selectedIndex);
            };

            list.makeItem = () => new Label();
            list.bindItem = (e, i) =>
            {
                var src = list.itemsSource as List<string>;
                ((Label)e).text = src != null && i < src.Count ? src[i] : "";
            };
            list.selectionChanged += _ => OnRowSelected();

            if (PlayerProgress.Instance != null)
                PlayerProgress.Instance.Wallet.OnGoldChanged += RefreshGold;

            RefreshGold(PlayerProgress.Instance != null
                ? PlayerProgress.Instance.Wallet.Gold : 0);
            SetOpen(false);
        }

        void OnDisable()
        {
            if (PlayerProgress.Instance != null)
                PlayerProgress.Instance.Wallet.OnGoldChanged -= RefreshGold;
        }

        void Update()
        {
            if (Input.GetKeyDown(toggleKey))
                SetOpen(!open);
        }

        void RefreshGold(int g)
        {
            string t = g + " G";
            if (goldLabel != null) goldLabel.text = t;
            if (goldHud != null) goldHud.text = t;
        }

        public void SetOpen(bool value)
        {
            if (GameFlow.IsInCutscene && value) return;

            open = value;
            if (backdrop != null)
                backdrop.style.display = open ? DisplayStyle.Flex : DisplayStyle.None;

            Time.timeScale = open ? 0f : 1f;

            if (open)
            {
                tabs.SetSelection(0);
                SelectTab(0);
            }
        }

        void SelectTab(int index)
        {
            tabIndex = index;
            contentTitle.text = Tabs[index];
            detail.text = "";
            actions.Clear();

            var rows = new List<string>();

            switch (index)
            {
                case 0: // Inventário — poções + armas
                    if (inventory != null)
                    {
                        foreach (var s in inventory.consumables)
                            rows.Add($"Poção x{s.count}  (+{s.item.healAmount} HP)");

                        for (int i = 0; i < inventory.ownedWeapons.Count; i++)
                        {
                            var w = inventory.ownedWeapons[i];
                            bool eq = equipment != null
                                && equipment.weaponInstance == w;
                            rows.Add($"{(eq ? "E " : "  ")}{w.weaponData.displayName}  "
                                + $"[{w.durability}/{w.maxDurability}]");
                        }
                    }
                    if (rows.Count == 0) rows.Add("(vazio)");
                    break;

                case 1: // Loja
                    if (shop != null && shop.Catalog != null)
                    {
                        foreach (var w in shop.Catalog.weapons)
                            if (w != null)
                                rows.Add($"{w.displayName}  — {w.buyPrice} G");
                        foreach (var c in shop.Catalog.consumables)
                            if (c != null)
                                rows.Add($"{c.displayName}  — {c.buyPrice} G");
                    }
                    if (rows.Count == 0) rows.Add("(loja vazia)");
                    break;

                case 2: // Ferreiro
                    if (inventory != null)
                    {
                        foreach (var w in inventory.ownedWeapons)
                            rows.Add($"{w.weaponData.displayName}  "
                                + $"[{w.durability}/{w.maxDurability}]  "
                                + $"reparar {w.RepairCost}G / estender {w.ExtendCost}G");
                    }
                    if (rows.Count == 0) rows.Add("(sem armas)");
                    break;

                case 3: // Opções
                    rows.Add("Reiniciar");
                    rows.Add("Sair do jogo");
                    break;
            }

            list.itemsSource = rows;
            list.Rebuild();
            list.ClearSelection();
        }

        void OnRowSelected()

                    }
                }
            }
            else if (tabIndex == 1 && shop != null && shop.Catalog != null)
            {
                int wCount = shop.Catalog.weapons.Count;

                {
                    var w = shop.Catalog.weapons[i];
                    detail.text = $"{w.description}\nPreço: {w.buyPrice} G";
                    var buy = new Button(() =>
                    {
                        shop.BuyWeapon(w);
                        SelectTab(1);

    {
        [SerializeField] UIDocument document;
        [SerializeField] VisualTreeAsset weaponRowTemplate; // WeaponListItem.uxml

        [Header("Systems")]
        [SerializeField] InventoryManager inventory;
        [SerializeField] EquipmentManager equipment;
        [SerializeField] BlacksmithService blacksmith;
        [SerializeField] ShopService shop;

        VisualElement root;
        ScrollView items;
        Label currentAttack;

        void OnEnable()
        {
            root = document.rootVisualElement;
            items = root.Q<ScrollView>("Items");
            currentAttack = root.Q<Label>("CurrentAttackDamage");

            var close = root.Q<Button>("CloseButton");
            if (close != null) close.clicked += Close;

            var weaponsTab = root.Q<Button>("WeaponsTabButton");
            if (weaponsTab != null) weaponsTab.clicked += ShowWeapons;
            // Shields / Armors / Accessory: placeholders por agora

            ShowWeapons();
        }

        public void Open()
        {
            root.style.display = DisplayStyle.Flex;
            Time.timeScale = 0f;
            ShowWeapons();
        }

        void Close()
        {
            root.style.display = DisplayStyle.None;
            Time.timeScale = 1f;
        }

        void ShowWeapons()
        {
            if (items == null) return;
            items.Clear();

            int equippedDamage = equipment != null && equipment.weaponInstance != null
                ? equipment.weaponInstance.weaponData.amount
                : 0;

            if (currentAttack != null)
                currentAttack.text = $"Current Attack Damage: {equippedDamage}";

            if (inventory == null) return;

            foreach (WeaponInstance wi in inventory.ownedWeapons)
                items.Add(BuildRow(wi, equippedDamage));
        }

        VisualElement BuildRow(WeaponInstance wi, int equippedDamage)
        {
            VisualElement row = weaponRowTemplate != null
                ? weaponRowTemplate.Instantiate()
                : new VisualElement();

            Weapon data = wi.weaponData;
            bool equipped = equipment != null && equipment.weaponInstance == wi;

            var name = row.Q<Label>("ItemName");
            var description = row.Q<Label>("Description");
            var indicator = row.Q("Indicator");
            var attention = row.Q("Attention");
            var status = row.Q<Label>("Status");
            var equipBtn = row.Q<Button>("EquipButton");
            var repairBtn = row.Q<Button>("RepairButton");
            var sellBtn = row.Q<Button>("SellButton");

            if (name != null) name.text = data.displayName;
            if (description != null) description.text = $"Attack Damage: {data.amount}";
            if (status != null)
                status.text = $"Current Durability: {wi.durability}/{wi.maxDurability}";

            // Indicator: verde se >= arma equipada, vermelho se pior
            if (indicator != null)
                indicator.style.backgroundColor = data.amount >= equippedDamage
                    ? new Color(0.35f, 0.86f, 0.35f)
                    : new Color(0.85f, 0.40f, 0.40f);

            // Attention: aviso quando durabilidade <= 25%
            if (attention != null)
                attention.style.display = wi.DurabilityNormalized <= 0.25f
                    ? DisplayStyle.Flex : DisplayStyle.None;

            if (equipBtn != null)
            {
                equipBtn.text = equipped ? "Equipped" : "Equip";
                equipBtn.SetEnabled(!equipped);
                equipBtn.clicked += () => { equipment.EquipWeapon(data); ShowWeapons(); };
            }

            if (repairBtn != null)
            {
                int cost = wi.RepairCost;
                repairBtn.text = cost > 0 ? $"Repair ({cost} Gold)" : "Repair";
                repairBtn.SetEnabled(cost > 0);
                repairBtn.clicked += () => { if (blacksmith.Repair(wi)) ShowWeapons(); };
            }

            if (sellBtn != null)
            {
                int value = shop != null ? shop.SellValue(data) : 0;
                sellBtn.text = $"Sell ({value} Gold)";
                sellBtn.clicked += () => { if (shop.SellWeapon(wi)) ShowWeapons(); };
            }

            return row;
        }
    }
}

[SerializeField] EquipmentManager equipment;
[SerializeField, Range(0f, 1f)] float sellRatio = 0.4f; // Sell = 40% do buyPrice

public int SellValue(Weapon weapon) =>
    weapon != null ? Mathf.RoundToInt(weapon.buyPrice * sellRatio) : 0;

public bool SellWeapon(WeaponInstance instance)
{
    if (instance == null || inventory == null) return false;
    if (PlayerProgress.Instance == null) return false;

    // desequipar se for a arma actual
    if (equipment != null && equipment.weaponInstance == instance)
        equipment.UnequipWeapon();

    if (!inventory.ownedWeapons.Remove(instance)) return false;

    PlayerProgress.Instance.Wallet.Add(SellValue(instance.weaponData));
    return true;
}

            Instance = this;
            var ve = document.rootVisualElement;
            root = ve.Q("root");
            dim = ve.Q("dim");
            slotTop = ve.Q("slot-top");
            slotCenter = ve.Q("slot-center");
            slotBottom = ve.Q("slot-bottom");
            boxTop = ve.Q("box-top");
            boxCenter = ve.Q("box-center");
            boxBottom = ve.Q("box-bottom");
            Hide();
        }

        void Update()
        {
            if (!playing) return;

            bool pressed = Input.GetMouseButtonDown(0)
                || Input.GetKeyDown(KeyCode.E)
                || Input.GetKeyDown(KeyCode.Space)
                || (Input.touchCount > 0 && Input.GetTouch(0).phase == TouchPhase.Began);

            if (!pressed) return;

            if (!waitingAdvance) skipping = true;
            else waitingAdvance = false;
        }

        public IEnumerator Play(DialogueSequence sequence)
        {
            if (sequence == null || sequence.lines == null) yield break;

            playing = true;
            root.style.display = DisplayStyle.Flex;

            for (int i = 0; i < sequence.lines.Count; i++)
            {
                DialogueSequence branch = null;
                yield return PlayLine(sequence.lines[i], b => branch = b);
                if (branch != null)
                {
                    yield return Play(branch);
                    break;
                }
            }

            Hide();
            playing = false;
        }

        IEnumerator PlayLine(DialogueLine line, System.Action<DialogueSequence> onBranch)
        {
            skipping = false;
            waitingAdvance = false;
            chosenIndex = -1;

            // Slot activo
            slotTop.RemoveFromClassList("active");
            slotCenter.RemoveFromClassList("active");
            slotBottom.RemoveFromClassList("active");

            VisualElement slot = slotBottom;
            VisualElement box = boxBottom;
            if (line.position == DialoguePosition.Top) { slot = slotTop; box = boxTop; }
            else if (line.position == DialoguePosition.Center) { slot = slotCenter; box = boxCenter; }
            slot.AddToClassList("active");

            // Fundo
            dim.EnableInClassList("visible", line.background == DialogueBackground.Dark
                || line.background == DialogueBackground.None);
            // Dark = dim + box dark; None = dim opcional + box transparente
            // Simplificação: Dark e None mostram dim; Box não.
            dim.EnableInClassList("visible", line.background != DialogueBackground.Box);

            box.Clear();
            box.RemoveFromClassList("bg-none");
            box.RemoveFromClassList("bg-dark");
            if (line.background == DialogueBackground.None) box.AddToClassList("bg-none");
            if (line.background == DialogueBackground.Dark) box.AddToClassList("bg-dark");

            // Portrait
            VisualElement portrait = null;
            if (line.portrait != null)
            {
                portrait = new VisualElement();
                portrait.AddToClassList("dlg-portrait");
                if (line.portraitSide == PortraitSide.Right)
                    portrait.AddToClassList("right");
                portrait.style.backgroundImage = new StyleBackground(line.portrait);
            }

            var textCol = new VisualElement();
            textCol.AddToClassList("dlg-textcol");

            var speaker = new Label(line.speakerName ?? "");
            speaker.AddToClassList("dlg-speaker");
            var body = new Label("");
            body.AddToClassList("dlg-body");
            var choicesRoot = new VisualElement();
            choicesRoot.AddToClassList("dlg-choices");
            var cont = new Label("▼");
            cont.AddToClassList("dlg-continue");
            cont.style.display = DisplayStyle.None;

            textCol.Add(speaker);
            textCol.Add(body);
            textCol.Add(choicesRoot);
            textCol.Add(cont);

            if (line.portraitSide == PortraitSide.Right)
            {
                box.Add(textCol);
                if (portrait != null) box.Add(portrait);
            }
            else
            {
                if (portrait != null) box.Add(portrait);
                box.Add(textCol);
            }

            // Typewriter
            string message = line.message ?? "";
            float delay = 1f / Mathf.Max(1f, charsPerSecond);
            for (int c = 0; c < message.Length; c++)
            {
                if (skipping) { body.text = message; break; }
                body.text += message[c];
                yield return new WaitForSecondsRealtime(delay);
            }

            if (line.choices != null && line.choices.Count > 0)
            {
                for (int i = 0; i < line.choices.Count; i++)
                {
                    int captured = i;
                    var btn = new Button(() => chosenIndex = captured)
                    {
                        text = line.choices[i].text
                    };
                    btn.AddToClassList("dlg-choice");
                    choicesRoot.Add(btn);
                }

                yield return new WaitUntil(() => chosenIndex >= 0);
                onBranch?.Invoke(line.choices[chosenIndex].next);
            }
            else
            {
                cont.style.display = DisplayStyle.Flex;
                waitingAdvance = true;
                yield return null; // evita que o mesmo tap avance
                yield return new WaitUntil(() => !waitingAdvance);
                onBranch?.Invoke(null);
            }
        }

        void Hide()
        {
            root.style.display = DisplayStyle.None;
            dim.RemoveFromClassList("visible");
            slotTop.RemoveFromClassList("active");
            slotCenter.RemoveFromClassList("active");
            slotBottom.RemoveFromClassList("active");
            boxTop.Clear();
            boxCenter.Clear();
            boxBottom.Clear();
        }
    }
}

8. Cutscenes — SequencePlayer

Porquê isto em vez de Moment + filhos EventBase
  • Dados vs. cena — diálogos/espera são ScriptableObjects reutilizáveis; mover NPC / controlos ficam como MonoBehaviour steps com refs de cena.
  • Ordem explícita — lista serializada no inspector, não depende de sibling order.
  • Cancelável + flag global — GameFlow.IsInCutscene bloqueia menu, lock-on, input.
  • Sem Timeline, sem Event Bus — só coroutines + UnityEvents locais se precisares.

using System.Collections;
using System.Collections.Generic;
using UnityEngine;
using UnityEngine.Events;

namespace AF
{
    /// <summary>
    /// Corre steps em ordem. Steps = componentes no mesmo GO (ou lista explícita).
    /// </summary>
    public class SequencePlayer : MonoBehaviour
    {
        [Tooltip("Se vazio, recolhe todos os MonoBehaviour ISequenceStep neste GO (ordem do inspector).")]
        [SerializeField] List<MonoBehaviour> stepBehaviours = new();

        public UnityEvent onStart;
        public UnityEvent onEnd;

        Coroutine running;
        public bool IsRunning { get; private set; }

        public void Trigger()
        {
            if (IsRunning) return;
            running = StartCoroutine(Run());
        }

        public void Cancel()
        {
            if (running != null) StopCoroutine(running);
            Finish();
        }

        IEnumerator Run()
        {
            IsRunning = true;
            GameFlow.SetCutscene(true);
            onStart?.Invoke();

            var steps = ResolveSteps();
            for (int i = 0; i < steps.Count; i++)
            {
                if (steps[i] != null)
                    yield return steps[i].Execute();
            }

            Finish();
        }

        List<ISequenceStep> ResolveSteps()
        {
            var list = new List<ISequenceStep>();
            if (stepBehaviours != null && stepBehaviours.Count > 0)
            {
                foreach (var mb in stepBehaviours)
                    if (mb is ISequenceStep s) list.Add(s);
            }
            else
            {
                foreach (var mb in GetComponents<MonoBehaviour>())
                    if (mb is ISequenceStep s && !ReferenceEquals(mb, this))
                        list.Add(s);
            }
            return list;
        }

        void Finish()
        {
            running = null;
            IsRunning = false;
            GameFlow.SetCutscene(false);
            onEnd?.Invoke();
        }

        void OnDisable()
        {
            if (IsRunning) Finish();
        }
    }
}

using System.Collections;
using UnityEngine;

namespace AF
{
    public class MoveCharacterStep : MonoBehaviour, ISequenceStep
    {
        [SerializeField] CharacterManager character;
        [SerializeField] Transform waypoint;
        [SerializeField] float timeout = 8f;

        public IEnumerator Execute()
        {
            if (character == null || waypoint == null) yield break;

            character.SetDestination(waypoint.position);
            float t = 0f;
            while (t < timeout)
            {
                var motor = character.Motor;
                if (motor != null)
                {
                    float dist = Vector3.Distance(
                        character.transform.position, waypoint.position);
                    if (dist <= motor.StoppingDistance + 0.15f)
                        yield break;
                }

                t += Time.deltaTime;
                yield return null;
            }
        }
    }
}
